# Python dependencies for Shingan's Python shims under `scripts/`.
# Install with:
#
#     pip install -r scripts/requirements-shim.txt
#
# The shims themselves only use the standard library, so the workers boot on
# any Python 3.10+ install. The dependencies below are required when actually
# parsing user files that import the relevant framework — without them, the
# worker stays alive but `parse_file` / `parse_content` return a
# "missing_<framework>" error.

# LangGraph parser (scripts/export_langgraph_server.py)
langgraph>=0.2.0

# CrewAI parser (scripts/export_crewai_server.py); requires Pydantic v2.
crewai>=0.50.0

# Optional but recommended (langgraph + crewai both depend on pydantic 2;
# pinned here so requirements stay self-contained for CI).
pydantic>=2.0
